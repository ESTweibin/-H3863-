#include "pinctrl.h"
#include "gpio.h"
#include "app_init.h"
#include "soc_osal.h"
#include "osal_debug.h"
#include "cmsis_os2.h"

#define SENSOR_TASK_STACK_SIZE      0x1000
#define SENSOR_TASK_PRIO            (osPriority_t)(25)

#define LIGHTING_SENSOR_PIN         0
#define BEEP_PIN                    1

void task_init(void)
{
    uapi_pin_set_mode(LIGHTING_SENSOR_PIN,PIN_MODE_0);
    uapi_pin_set_mode(BEEP_PIN,PIN_MODE_0);
    uapi_gpio_set_dir(LIGHTING_SENSOR_PIN,GPIO_DIRECTION_INPUT);
    uapi_gpio_set_dir(BEEP_PIN,GPIO_DIRECTION_OUTPUT);
    uapi_pin_set_pull(LIGHTING_SENSOR_PIN,PIN_PULL_TYPE_UP);
}

void *lighting_sensor_task(const char *arg)
{
    UNUSED(arg);

    task_init();

    while (1)
    {
        if(uapi_gpio_get_val(LIGHTING_SENSOR_PIN) == GPIO_LEVEL_HIGH)
        {
            uapi_gpio_set_val(BEEP_PIN,GPIO_LEVEL_HIGH);   
        }
        else
        {
            uapi_gpio_set_val(BEEP_PIN,GPIO_LEVEL_LOW);
        }
        osal_msleep(1000);
    }
    
    return NULL;
}

static void light_sensor_entry(void)
{
    osThreadAttr_t attr;

    attr.name = "LightSensorTask";
    attr.attr_bits = 0U;
    attr.cb_mem = NULL;
    attr.cb_size = 0U;
    attr.stack_mem = NULL;
    attr.stack_size = SENSOR_TASK_STACK_SIZE;
    attr.priority = SENSOR_TASK_PRIO;

    if (osThreadNew((osThreadFunc_t)lighting_sensor_task, NULL, &attr) == NULL) {
        osal_printk("Error: Failed to create LightSensorTask.\r\n");
    }
}

// 注册并运行入口函数
app_run(light_sensor_entry);
